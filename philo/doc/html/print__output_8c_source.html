81void print_output(t_philo *philo, bool monitor, t_status status)
82{
83 pthread_mutex_lock(&philo->data->print_lock);
84 if (!all_philo_is_alive(philo->data) && !monitor)
85 {
86 pthread_mutex_unlock(&philo->data->print_lock);
87 return ;
88 }
89 if (status == DIE)
90 print_status(philo, "died", RED);
91 else if (status == EAT)
92 print_status(philo, "is eating", PURPLE);
93 else if (status == SLEEP)
94 print_status(philo, "is sleeping", CYAN);
95 else if (status == THINK)
96 print_status(philo, "is thinking", GREEN);
97 else if (status == FORK_1 || status == FORK_2)
98 print_status(philo, "has taken a fork", ORANGE);
99 pthread_mutex_unlock(&philo->data->print_lock);
100}
