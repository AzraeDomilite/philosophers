23static pthread_mutex_t *init_forks(t_data *data)
24{
25 pthread_mutex_t *forks;
26 size_t index;
27
28 forks = malloc(sizeof(pthread_mutex_t) * data->nb_of_philos);
29 if (!forks)
30 return (null_error_msg(MALLOC_ERROR));
31 index = 0;
32 while (index < data->nb_of_philos)
33 {
34 if (pthread_mutex_init(&forks[index], 0) != 0)
35 return (null_error_msg(MUTEX_ERROR));
36 index++;
37 }
38 return (forks);
39}
