27static void philo_think(t_philo *philo, bool first)
28{
29 time_t time_to_think;
30
31 pthread_mutex_lock(&philo->meal_time_lock);
32 time_to_think = (philo->data->time_to_die
33 - (get_time() - philo->last_meal_time)
34 - philo->data->time_to_eat) / 2;
35 pthread_mutex_unlock(&philo->meal_time_lock);
36 if (time_to_think < 0)
37 time_to_think = 0;
38 if (time_to_think == 0 && first == true)
39 time_to_think = 1;
40 if (time_to_think > 600)
41 time_to_think = 200;
42 print_output(philo, false, THINK);
43 philo_sleep(philo->data, time_to_think);
44}

134void *philo(void *data)
135{
136 t_philo *philo;
137
138 philo = (t_philo *)data;
139 if (!philo->data->nb_time_ph_must_eat)
140 return (NULL);
141 pthread_mutex_lock(&philo->meal_time_lock);
142 philo->last_meal_time = philo->data->start_timer;
143 pthread_mutex_unlock(&philo->meal_time_lock);
144 start_delay(philo->data->start_timer);
145 if (!philo->data->time_to_die)
146 return (NULL);
147 if (philo->data->nb_of_philos == 1)
148 return (one_philo(philo));
149 if (philo->id % 2)
150 philo_think(philo, true);
151 if (philo->data->nb_of_philos - 1 == philo->id
152 && philo->data->nb_of_philos % 2)
153 philo_think(philo, true);
154 while (all_philo_is_alive(philo->data))
155 {
156 philo_eat_sleep(philo);
157 philo_think(philo, false);
158 }
159 return (NULL);
160}
