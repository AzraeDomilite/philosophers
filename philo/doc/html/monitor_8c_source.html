58static bool end(t_data *data)
59{
60 size_t index;
61 size_t nb_philo_reach_max_meals;
62
63 index = 0;
64 nb_philo_reach_max_meals = 0;
65 while (index < (size_t)data->nb_of_philos)
66 {
67 pthread_mutex_lock(&data->philos[index]->meal_time_lock);
68 if (philo_is_dead(data->philos[index]))
69 return (true);
70 if (data->nb_time_ph_must_eat != -1)
71 if ((data->philos[index]->nb_meal_eat)
72 >= (size_t)data->nb_time_ph_must_eat)
73 nb_philo_reach_max_meals++;
74 pthread_mutex_unlock(&data->philos[index]->meal_time_lock);
75 index++;
76 }
77 if (nb_philo_reach_max_meals == (size_t)data->nb_of_philos
78 && data->nb_time_ph_must_eat != -1)
79 return (update_sim_status(data, true), true);
80 return (false);
81}
